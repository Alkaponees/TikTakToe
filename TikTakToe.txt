int get_move()//введення числа та перевірка чи воно правильне
{
	system("cls");
	Navigator();
	View_Board();
	int move;
	cin >> move;
	while (move > 9 || move < 1 || board[move - 1] != '-') //перевірка введеного числа
	{
		cout << "Enter move again (1-9)\n";
		cin >> move;
	}
	return move;
}
void play()//игра
{
	
	for (int i = 0; i < 5; i++)
	{
		int move = get_move();
		get_move();
		if (i % 2 == 0)
		{
			board[move - 1] = 'X';

		}
		else
		{
			board[move - 1] = 'O';

		}
	}
		
}
